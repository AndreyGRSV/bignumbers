360 bdig& div(const bdig& v, bdig* premainder = NULL)
361 {
362
363 if (!v)
364 return *this;
365
366 CRTHEAPOBJNP(remainder);
367 CRTHEAPOBJNP(remainder_tmp);
368 CRTHEAPOBJNP(borrow);
369 CRTHEAPOBJ(divisor, v);
370 CRTHEAPOBJ(result, 0);
371
372 bool store_is_negative = is_negative;
373 is_negative = false;
374 *remainder = *this;
375 integer.clear();
376
377 *divisor = (*divisor).abs();
378 // Shift Divider to Value max significant index
379 const std::size_t ri = (*remainder).most_significant_index();
380 const std::size_t di = (*divisor).most_significant_index();
381 std::size_t shift = 0;
382 if (ri < di) {
383 shift = di - ri;
384 (*divisor).shlb(shift);
385 }
386 // If Divider > Value Shit Divider back to one position(In current sample is not needeed)
387 if ((*divisor) > (*remainder) && shift) {
388 (*divisor).shrb(1);
389 shift--;
390 }
391
392 // Both items have value only in one first limb
393 std::size_t index = isz - 1;
394 if (ri == index && di == index) {
395 (*result).integer.set(index, (*remainder).integer[index] / (*divisor).integer[index]);
396 (*remainder).integer.set(index, (*remainder).integer[index] % (*divisor).integer[index]);
397 } else {
398 const upT BaseValue = static_cast<upT>(std::numeric_limits<T>::max()) + 1;
399
400 while ((*divisor) <= (*remainder)) {
401
402 upT result_value = 0;
403 upT last_diff = 0;
404 const std::size_t result_index = (*divisor).most_significant_index();
405 const std::size_t remainder_index = (*remainder).most_significant_index();
406 std::size_t current_index = result_index;
407 while (current_index < isz) {
408 if (result_index == current_index) {
409 result_value = (*remainder).integer[current_index];
410 if (remainder_index < result_index) {
411 result_value += (*remainder).integer[current_index - 1] * BaseValue;
412 }
413 result_value -= (*borrow).integer[current_index];
414 upT remainder_value = result_value;
415 result_value /= (*divisor).integer[current_index];
416 if (result_value >= BaseValue)
417 result_value = BaseValue - 1;
418 remainder_value -= (*divisor).integer[current_index] * result_value;
419 (*remainder_tmp).integer.set(current_index, static_cast<T>(remainder_value));
420 current_index++;
421 } else {
422 upT need_value = (*divisor).integer[current_index] * result_value;
423 upT borrow_value = 0;
424 if (need_value > (*remainder).integer[current_index]) {
425 upT full_value = need_value + (*borrow).integer[current_index] - (*remainder).integer[current_index];
426 borrow_value = full_value / BaseValue + ((full_value % BaseValue) ? 1 : 0);
427 }
428 upT diff = borrow_value < (*borrow).integer[current_index - 1] ?
429 (*borrow).integer[current_index - 1] - borrow_value :
430 borrow_value - (*borrow).integer[current_index - 1];
431 if (diff && diff == last_diff) {
432 if ((*borrow).integer[current_index - 1] != borrow_value) {
433 (*borrow).integer.set(current_index - 1, static_cast<T>(borrow_value));
434 }
435
436 (*remainder).div_bits((*divisor), remainder_tmp.get());
437 result_value = (*remainder).integer[isz - 1];
438 break;
439 }
440 last_diff = diff;
441 if (borrow_value != (*borrow).integer[current_index - 1]) {
442 (*borrow).integer.set(current_index - 1, static_cast<T>(borrow_value));
443 current_index--;
444 } else {
445 upT remainder_value = (*borrow).integer[current_index - 1] * BaseValue
446 + (*remainder).integer[current_index]
447 - (*borrow).integer[current_index]
448 - (*divisor).integer[current_index] * result_value;
449 if (remainder_value >= BaseValue) {
450 (*remainder_tmp).integer.set(current_index - 1, (*remainder_tmp).integer[current_index - 1] + 1);
451 }
452 (*remainder_tmp).integer.set(current_index, static_cast<T>(remainder_value));
453 current_index++;
454 }
455 }
456 }
457
458 (*result).integer.set(isz - 1 - shift, static_cast<T>(result_value));
459 (*remainder) = (*remainder_tmp);
460 (*remainder_tmp) = 0;
461 (*borrow) = 0;
462
463 if (!shift)
464 break;
465
466 shift--;
467 (*divisor).shrb(1);
468 }
469 }
470
471 is_negative = store_is_negative != v.is_negative;
472 (*remainder).is_negative = store_is_negative != v.is_negative;
473 if (premainder)
474 *premainder = *remainder;
475
476 (*result).is_negative = store_is_negative != v.is_negative;
477 if (premainder != this)
478 *this = *result;
479 return *this;
480 }

506 bdig& div_bits(const bdig& v, bdig* premainder = NULL)
507 {
508
509 if (!v)
510 return *this;
511
512 CRTHEAPOBJNP(remainder);
513 CRTHEAPOBJ(divisor, v);
514
515 bool store_is_negative = is_negative;
516 is_negative = false;
517 *remainder = *this;
518 integer.clear();
519
520 CRTHEAPOBJNP(shift);
521 (*shift).integer.set(isz - 1, 1);
522 *divisor = (*divisor).abs();
523
524 // Align divisor with remainder in array
525 std::size_t ri = (*remainder).most_significant_index();
526 std::size_t di = (*divisor).most_significant_index();
527 if (ri < di) {
528 (*divisor).shlb(di - ri);
529 (*shift).shlb(di - ri);
530 }
531
532 while (*remainder >= v) {
533 while (!(*remainder).integer[ri] && ri < isz) {
534 if ((*divisor).integer[ri]) {
535 int d_msb = most_significant_bit((*divisor).integer[ri]);
536 (*divisor).shr(d_msb + 1);
537 (*shift).shr(d_msb + 1);
538 }
539 ri++;
540 }
541 int d_msb = most_significant_bit((*divisor).integer[ri]);
542 int r_msb = most_significant_bit((*remainder).integer[ri]);
543 int bits = r_msb - d_msb;
544
545 std::size_t msi_divisor = (*divisor).most_significant_index();
546 std::size_t msi_shift = (*shift).most_significant_index();
547 std::size_t msi = std::min(msi_shift, msi_divisor);
548 msi -= msi ? 1 : 0;
549
550 if (bits) {
551 T set_bit = 0, set_bit1 = 0;
552 if (bits > 0) {
553 for (std::size_t i = isz - 1; i >= msi; i--) {
554 set_bit = (*divisor).shl_sop(i, bits, set_bit);
555 set_bit1 = (*shift).shl_sop(i, bits, set_bit1);
556 if (!i)
557 break;
558 }
559 } else {
560 for (std::size_t i = msi; i < isz; ++i) {
561 set_bit = (*divisor).shr_sop(i, -bits, set_bit);
562 set_bit1 = (*shift).shr_sop(i, -bits, set_bit1);
563 }
564 }
565 }
566 while ((*divisor) > (*remainder)) {
567 T set_bit = 0, set_bit1 = 0;
568 for (std::size_t i = msi; i < isz; i++) {
569 set_bit = (*divisor).shr_sop(i, 1, set_bit);
570 set_bit1 = (*shift).shr_sop(i, 1, set_bit1);
571 }
572 }
573
574 std::size_t msi_quotient = most_significant_index();
575 std::size_t msi_remainder = (*remainder).most_significant_index();
576 msi = std::min(msi_remainder, msi_quotient);
577 msi -= msi ? 1 : 0;
578
579 bool cs = false, csm = false;
580 for (std::size_t i = isz - 1; i >= msi; --i) {
581 csm = sum_sop(i, *shift, csm);
582 cs = (*remainder).sub_sop(i, *divisor, cs);
583 if (!i)
584 break;
585 }
586 if (!*divisor)
587 break;
588 }
589 is_negative = store_is_negative != v.is_negative;
590 (*remainder).is_negative = store_is_negative != v.is_negative;
591 if (premainder)
592 *premainder = *remainder;
593 return *this;
594 }
